CREATE DATABASE alquranlearningcenter;

CREATE TABLE users (
	id int NOT NULL AUTO_INCREMENT,
	uid varchar(255) NOT NULL,
	fname varchar(255) NOT NULL,
	lname varchar(255) NOT NULL,
	email varchar(255) NOT NULL,
	password varchar(255) NOT NULL,
	is_confirmed varchar(255) DEFAULT 'false',
	age varchar(255) NOT NULL,
	languages varchar(255) DEFAULT 'ENGLISH',
	gender varchar(255) NOT NULL,
	proficiency varchar(255) NOT NULL,
	availability varchar(255) NOT NULL,
	is_enabled varchar(255) DEFAULT 'false',
	account_type varchar(255) DEFAULT 'student',
	created TIMESTAMP DEFAULT NOW(),
	PRIMARY KEY(id)
);

CREATE TABLE sessions (
	id int NOT NULL AUTO_INCREMENT,
	sid varchar(255) NOT NULL,
	uid_teacher varchar(255) NOT NULL,
	uid_student varchar(255) NOT NULL,
	datetime varchar(255) NOT NULL,
	duration varchar(255) NOT NULL,
	created TIMESTAMP DEFAULT NOW(),
	is_enabled varchar(255) DEFAULT 'false',
	PRIMARY KEY(id)
);

CREATE TABLE messages (
	id int NOT NULL AUTO_INCREMENT,
	mid varchar(255) NOT NULL,
	uid_from varchar(255) NOT NULL,
	uid_to varchar(255) NOT NULL,
	subject varchar(255) NOT NULL,
	message text NOT NULL,
	sent TIMESTAMP DEFAULT NOW(),
	PRIMARY KEY(id)
);


CREATE TABLE announcements (
	id int NOT NULL AUTO_INCREMENT,
	aid varchar(255) NOT NULL,
	to varchar(255) DEFAULT 'everyone',
	is_enabled varchar(255) DEFAULT 'true',
	PRIMARY KEY(id)	
);


CREATE TABLE homework (
	id int NOT NULL AUTO_INCREMENT,
	hid varchar(255) NOT NULL,
	uid_teacher varchar(255) NOT NULL,
	uid_student varchar(255) NOT NULL,
	content text NOT NULL,
	grade varchar(255) NOT NULL,
	sent TIMESTAMP DEFAULT NOW(),
	PRIMARY KEY(id)
);
